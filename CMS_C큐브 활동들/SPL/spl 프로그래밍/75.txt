

void setup()
{
    NSBegin(64, 2)
	MatrixClear()
}

void loop()
{
    MatrixClear(
	X=random(8)
	Y=random(8)
    if(X>=Y)
	 {
	    for(i=X;i<8;i=i+1)
		  {
		      for(j=Y;j<Y+8-X;j=j+1)
			    {
				    MatrixDrawPixel(i,j,0,0,1)
					MatrixWriteDisplay()
			    }
		  }
     }
    if(Y>=X)
	 {
	    for(i=Y;i<8;i=i+1)
		  {
		      for(j=X;j<X+8-Y;j=j+1)
			    {
				    MatrixDrawPixel(i,j,0,0,1)
					MatrixWriteDisplay()
			    }
		  }
     }
Delay(1000)
}
