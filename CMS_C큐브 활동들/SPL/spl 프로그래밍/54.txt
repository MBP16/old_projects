
void setup()
{
	LcdBackLight()
	LcdClear()
}
void loop()
{   
	a= AnalogRead(0)
    if(a>950&&a<970)
     {
	    LcdPrint("A")
     }
    if(a>500&&a<520)
	 {
	    LcdPrint("B")
     }
    if(a>870&&a<890)
	 {
	    LcdPrint("C")
	 }
	if(a>670&&a<690)
	 {
	    LcdPrint("D")
	 }
    if(a>760&&a<780)
	 {
	    LcdClear()
	 }
   Delay(200)
}
