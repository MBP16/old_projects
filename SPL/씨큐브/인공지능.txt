int GO_CNT =0


void setup()
{
	
}

void loop()
{
	a0= AnalogRead(0)
	a1= AnalogRead(1)
	a2= AnalogRead(2)
	if(a0<300||a1<300)
	{
		DriveWrite(-150,150)
		int d= random(300,500)
		Delay(d)
		GO_CNT =10
	}
	else if(GO_CNT>0)
	{
		GO_CNT=GO_CNT-1
		DriveWrite(150,150)
		Delay(100)
	}
	else if(a2>100)
	{
		DriveWrite(150,150)
		Delay(100)
	}
	else
	{
		while(a2 <=200)
		{
			DriveWrite(-150,150)
			Delay(100)
			a2= AnalogRead(2)
		}
	}
}
