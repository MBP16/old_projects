
void setup()
{
}

void loop()
{
	NSBegin(8, 3)
	NSClear()
	for (i = 0; i <= 176; i=i+22)
	{
		ServoWrite(2, i)
		for (a = 0; a <= 210; a=a+30)
		{
			NSSetColor(a/30, 255, a, 0)
			NSShow()
			Delay(300)
		}
		Delay(300)
	}
	for (j = 176; j >= 0; j=j-22)
	{
		ServoWrite(2, j)
		NSSetColor(j/22, 0, 0, 0)
		NSShow()
		Delay(300)
	}
}
