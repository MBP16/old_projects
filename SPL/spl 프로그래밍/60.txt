

void setup()
{
	MatrixClear()
}

void loop()
{
    MatrixClear()
	for (i = 0; i < 5; i++)
	  {
		  MatrixDrawPixel(i, 1, LED_GREEN)
		  MatrixDrawPixel(6, i, LED_RED)
		  MatrixDrawPixel(1, 7-i, LED_YELLOW)
		  MatrixDrawPixel(7-i, 6, LED_GREEN)
		  MatrixWriteDisplay()
		  Delay(1000)
	  }
    MatrixWriteDisplay()
}
