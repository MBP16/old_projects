

void setup()
{
	NSBegin(64,2)
	MatrixClear()
}

void loop()
{
    MatrixClear()
    X=random(8)
	Y=random(8)
    for(i=0;i<8;i=i+1)
	  {
	     MatrixDrawPixel(X,Y,120,120,0)
		 MatrixDrawPixel(X,i,LED_GREEN)
		 MatrixWriteDisplay()
	  }
	for(j=0;j<8;j=j+1)
	  {
	     MatrixDrawPixel(j,Y,LED_RED)
	     MatrixDrawPixel(X,Y,255,255,0)
		 MatrixWriteDisplay()
	  }
    Delay(1000)
}
