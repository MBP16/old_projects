

void setup()
{
	LcdBackLight()
	LcdClear()
}

void loop()
{
	LcdClear()
	DigitalTempRead(3)
	t = Temperature
	h = Humidity
	LcdHome()
	LcdPrint("T")
	LcdPrint(":")
	LcdPrint(t)
	LcdSetCursor(0, 1)
	LcdPrint("H")
	LcdPrint(":")
	LcdPrint(h)
	a=180-6*(t-10)
	ServoWrite(2,a)
	Delay(100)
}
