
void setup()
{
	NSBegin(64 ,2)
	NSClear()
}


void loop()
{
	for(i=0;i<64;i=i+8)
	  {
	      for(k=-1;k>-8;k=k-1)
		     {
			    NSClear(i+j,0,0,0)
				NSShow()
			 }
	      for(j=0;j<8 ;j=j+1)
		    {
			    NSSetColor(i+j,120,0,0)
			    NSShow()
		    }
		  Delay(500)
	  }
}
