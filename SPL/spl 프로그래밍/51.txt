

void setup()
{
	LcdBackLight()
	LcdClear()		
}
b=0
c=0
void loop()
{   
    LcdClear()
	a= AnalogRead(0)
	if (a>950&&a<970)
	 {
		b=b+1
	 }
	if (a>500&&a<520)
	 {
	    c=b+c
		b=0
     }
    LcdHome()
    LcdPrint(b)
	LcdSetCursor(0, 1)
	LcdPrint(c)
	Delay(200)
}
