
void setup()
{
	ServoWrite(2, 0)
	ServoWrite(3, 0)
	LcdBackLight()
}

void loop()
{
	for (i = 0; i <= 36; i=i+1)
	{
		LcdClear()
		ServoWrite(2, 5*i)
		LcdSetCursor(0, 0)
		LcdPrint(5*i)
		for (j = 1; j <= 180; j=j+1)
		{
			ServoWrite(3, j)
			LcdSetCursor(0, 1)
			LcdPrint(j)
			Delay(100)
		}
	}
}
