#include<analogkeypad.h>
int PX[8]
int PY[8]
i=0
int F[14]={587,784,587,784,587,784,587,784,587,784,587,784,587,784,}
int f=-1
void playsound()
{
	f=f+1
	f=f%14
	Tone(3,F[f],80)
	Delay(100)
}
void setup()   
{
	NSBegin(64,2)
	MatrixClear()
	for(i=0;i< 8;i++)
	{
		PX[i]=random(8)
		PY[i]=-1-i
	}
}

x=4
void loop()
{
	int a=KeyPad(0)
	MatrixDrawPixel(x,7,LED_YELLOW)
	MatrixWriteDisplay()
	if(a==1&&x>0&&x<=7)
	{
		x=x-1
	}
	if(a==2&&x>=0&&x<7)
	{
		x=x+1
	}
	for(int i = 0; i <= 7 ; i = i+1)
	{
		PY[i]=PY[i]+1
		if (PY[i]>7)
		{
			PY[i]=0
			PX[i]=random(8)
		}
		MatrixDrawPixel(PX[i],PY[i],LED_RED)
	}
	MatrixWriteDisplay()
	Delay(200)
	MatrixClear()
	for(f=1;f<i;f=f+1)
	{
		if(PX[f]==x&&PY[f]==7)
		{
			for(j=0;j<8;j=j+1)
			{
				for(k=0;k<8;k=k+1)
				{
					MatrixDrawPixel(j,k,LED_RED)
					MatrixWriteDisplay()
					playsound()
				}
			}
			
		}
	}
}
