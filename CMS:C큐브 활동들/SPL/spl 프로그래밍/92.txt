int y=0
int x=0
int c=0
int F[14]={660,660,524,660,784,392,524,392,328,440,492,466,440,392}
int f=-1

void playsound()
 {
     f=f+1
	 f=f%14
     Tone(9,F[f],80)
	 Delay(100)
}

void setup()
{
	
}

void loop()
{
    playsound()
    for(int i=0;i<10;i=i+1)
	  {
	       x=random(8)
		   y=random(8)
		   c=random(1,4)
		   MatrixDrawPixel(x,y,c)
	  }
	  MatrixWriteDisplay()
	  Delay(1)
	  MatrixClear()
	  MatrixWriteDisplay()
	  Delay(1)	
}
