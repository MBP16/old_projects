#include<analogkeypad.h>

int PX[8]
int PY[8]
int F[14]={660,660,524,660,784,392,524,392,328,440,492,466,440,392}
int f=-1

void setup()
{
    NSBegin(64,2)
	for(i=0;i< 8;i++)
	{
		PX[i]=random(8)
		PY[i]=-1-i
	}
}
void loop()
{
	for(int i = 0; i <= 7 ; i = i+1)
	{
		PY[i]=PY[i]+1
		if (PY[i]>7)
		{
			PY[i]=0
			PY[i]=random(8)
		}
		MatrixDrawPixel(PX[i],PY[i],LED_RED)
		Delay(100)
	}
	MatrixWriteDisplay()
	MatrixClear()
}
